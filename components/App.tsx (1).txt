import React, { useState } from 'react';
import { Header } from './components/Header';
import { AdUnit } from './components/AdUnit';
import { LogoStyle, GenerationSettings, GeneratedLogo } from './types';
import { geminiService } from './services/geminiService';
import { APP_NAME } from './constants';
import JSZip from 'jszip';

const App: React.FC = () => {
  const [settings, setSettings] = useState<GenerationSettings>({
    prompt: '',
    style: LogoStyle.MINIMALIST,
    aspectRatio: '1:1'
  });
  
  const [isGenerating, setIsGenerating] = useState(false);
  const [logos, setLogos] = useState<GeneratedLogo[]>([]);
  const [selectedLogo, setSelectedLogo] = useState<GeneratedLogo | null>(null);

  const handleGenerate = async () => {
    if (!settings.prompt.trim()) return;
    setIsGenerating(true);
    try {
      const url = await geminiService.generateLogo(settings);
      const newLogo: GeneratedLogo = {
        id: crypto.randomUUID(),
        url,
        prompt: settings.prompt,
        timestamp: Date.now(),
        style: settings.style
      };
      setLogos(prev => [newLogo, ...prev]);
      setSelectedLogo(newLogo);
    } catch (err) {
      alert("Generation failed. Please check your API key.");
    } finally {
      setIsGenerating(false);
    }
  };

  const handleDownloadPNG = (url: string, id: string) => {
    const link = document.createElement('a');
    link.href = url;
    link.download = `logo-${id.slice(0, 5)}.png`;
    link.click();
  };

  const handleExportZIP = async () => {
    if (logos.length === 0) return;
    const zip = new JSZip();
    logos.forEach((logo, index) => {
      const base64Data = logo.url.split(',')[1];
      zip.file(`logo-${index + 1}.png`, base64Data, { base64: true });
    });
    const content = await zip.generateAsync({ type: 'blob' });
    const url = URL.createObjectURL(content);
    const link = document.createElement('a');
    link.href = url;
    link.download = 'brand-assets.zip';
    link.click();
    URL.revokeObjectURL(url);
  };

  const copyToClipboard = (text: string) => {
    navigator.clipboard.writeText(text);
    alert("Prompt copied to clipboard!");
  };

  return (
    <div className="min-h-screen flex flex-col bg-slate-50">
      <Header />
      
      <main className="flex-grow max-w-7xl mx-auto w-full px-4 sm:px-6 py-10 grid grid-cols-1 lg:grid-cols-12 gap-8">
        
        {/* Sidebar */}
        <aside className="lg:col-span-4 space-y-6">
          <div className="bg-white rounded-3xl p-6 shadow-sm border border-slate-200/60">
            <h2 className="text-xl font-bold text-slate-900 mb-6">Logo Creator</h2>
            
            <div className="space-y-5">
              <div>
                <label className="block text-[10px] font-black text-slate-400 uppercase tracking-widest mb-2">Description</label>
                <textarea 
                  className="w-full bg-slate-50 border border-slate-100 rounded-xl p-3 text-sm focus:ring-2 ring-indigo-500 outline-none resize-none"
                  placeholder="What's your business about?"
                  rows={3}
                  value={settings.prompt}
                  onChange={(e) => setSettings({...settings, prompt: e.target.value})}
                />
              </div>

              <div>
                <label className="block text-[10px] font-black text-slate-400 uppercase tracking-widest mb-2">Style</label>
                <div className="grid grid-cols-2 gap-2 max-h-48 overflow-y-auto pr-1 custom-scrollbar">
                  {Object.values(LogoStyle).map(style => (
                    <button 
                      key={style}
                      onClick={() => setSettings({...settings, style})}
                      className={`py-2.5 px-2 rounded-lg text-[10px] font-bold uppercase transition-all ${
                        settings.style === style ? 'bg-indigo-600 text-white shadow-md shadow-indigo-100' : 'bg-slate-50 text-slate-500 hover:bg-slate-100'
                      }`}
                    >
                      {style}
                    </button>
                  ))}
                </div>
              </div>

              <button 
                onClick={handleGenerate}
                disabled={isGenerating || !settings.prompt}
                className={`w-full py-4 rounded-xl font-black text-white text-xs tracking-widest transition-all ${
                  isGenerating ? 'bg-slate-300 animate-pulse cursor-not-allowed' : 'logo-gradient hover:opacity-90 active:scale-95 shadow-lg shadow-indigo-100'
                }`}
              >
                {isGenerating ? "GENERATING..." : "CREATE LOGO"}
              </button>
            </div>
          </div>

          <AdUnit slot="SIDEBAR_AD" className="h-[250px] w-full" />
        </aside>

        {/* Workspace */}
        <section className="lg:col-span-8 space-y-6">
          {selectedLogo ? (
            <div className="animate-fade-in space-y-6">
              <div className="bg-white p-6 sm:p-10 rounded-[2rem] shadow-sm border border-slate-200/60">
                <div className="flex flex-col sm:flex-row items-center justify-between gap-4 mb-8">
                   <h3 className="text-sm font-black uppercase tracking-widest text-slate-400">Master Logo</h3>
                   <div className="flex gap-2 w-full sm:w-auto">
                      <button 
                        onClick={() => handleDownloadPNG(selectedLogo.url, selectedLogo.id)}
                        className="flex-1 sm:flex-none bg-indigo-600 text-white px-5 py-2.5 rounded-xl text-[10px] font-black uppercase hover:bg-indigo-700 transition-all shadow-md shadow-indigo-50"
                      >
                        PNG
                      </button>
                      <button 
                        onClick={() => copyToClipboard(selectedLogo.prompt)}
                        className="flex-1 sm:flex-none bg-slate-100 text-slate-600 px-5 py-2.5 rounded-xl text-[10px] font-black uppercase hover:bg-slate-200 transition-all"
                      >
                        Copy Prompt
                      </button>
                      {logos.length > 0 && (
                        <button 
                          onClick={handleExportZIP}
                          className="flex-1 sm:flex-none bg-slate-900 text-white px-5 py-2.5 rounded-xl text-[10px] font-black uppercase hover:bg-black transition-all"
                        >
                          ZIP ({logos.length})
                        </button>
                      )}
                   </div>
                </div>

                <div className="grid grid-cols-1 md:grid-cols-2 gap-10">
                  <div className="aspect-square bg-white rounded-3xl border border-slate-100 flex items-center justify-center p-8 shadow-inner relative group">
                    <img src={selectedLogo.url} className="max-w-full max-h-full object-contain drop-shadow-xl" alt="Preview" />
                  </div>
                  
                  <div className="space-y-4">
                    <div className="bg-slate-900 aspect-video rounded-2xl flex items-center justify-center p-6 relative overflow-hidden group">
                       <div className="absolute inset-0 bg-[url('https://images.unsplash.com/photo-1497366216548-37526070297c?auto=format&fit=crop&q=80&w=1000')] bg-cover opacity-10 grayscale"></div>
                       <img src={selectedLogo.url} className="w-12 h-12 object-contain relative z-10 brightness-0 invert opacity-80" />
                       <span className="absolute bottom-2 left-3 text-[7px] text-white/40 font-black uppercase tracking-widest">Office Mockup</span>
                    </div>
                    <div className="bg-indigo-50 aspect-video rounded-2xl flex items-center justify-center p-6 relative">
                       <img src={selectedLogo.url} className="w-10 h-10 object-contain shadow-2xl" />
                       <span className="absolute bottom-2 left-3 text-[7px] text-indigo-300 font-black uppercase tracking-widest">Digital Brand</span>
                    </div>
                  </div>
                </div>
              </div>

              {logos.length > 1 && (
                <div className="space-y-3">
                  <h4 className="text-[10px] font-black text-slate-400 uppercase tracking-widest ml-1">History</h4>
                  <div className="flex gap-3 overflow-x-auto pb-4 custom-scrollbar">
                    {logos.map(logo => (
                      <button 
                        key={logo.id} 
                        onClick={() => setSelectedLogo(logo)}
                        className={`flex-shrink-0 w-20 h-20 bg-white rounded-xl p-3 border-2 transition-all ${
                          selectedLogo.id === logo.id ? 'border-indigo-600 shadow-lg' : 'border-slate-100 opacity-60 hover:opacity-100 hover:border-slate-200'
                        }`}
                      >
                        <img src={logo.url} className="w-full h-full object-contain" />
                      </button>
                    ))}
                  </div>
                </div>
              )}
            </div>
          ) : (
            <div className="h-[400px] sm:h-[500px] bg-white rounded-[2.5rem] border-2 border-dashed border-slate-200 flex flex-col items-center justify-center text-center p-8">
              <div className="w-20 h-20 logo-gradient rounded-3xl flex items-center justify-center text-white text-3xl mb-6 shadow-xl shadow-indigo-100">AI</div>
              <h3 className="text-2xl font-black text-slate-900 mb-2">Ready to Start?</h3>
              <p className="text-slate-400 max-w-xs text-sm font-medium">Describe your vision and watch the magic happen.</p>
            </div>
          )}
          
          <AdUnit slot="FOOTER_AD" className="h-32 w-full" />
        </section>
      </main>

      <footer className="bg-white border-t border-slate-100 py-10 px-6 mt-10">
        <div className="max-w-7xl mx-auto flex flex-col md:flex-row items-center justify-between gap-6">
          <div className="flex items-center gap-2">
            <div className="w-6 h-6 logo-gradient rounded-md"></div>
            <span className="font-black text-sm uppercase tracking-tighter">{APP_NAME}</span>
          </div>
          <div className="flex gap-6 text-[10px] font-black uppercase tracking-widest text-slate-400">
             <a href="#" className="hover:text-indigo-600 transition-colors">Privacy</a>
             <a href="#" className="hover:text-indigo-600 transition-colors">Terms</a>
             <a href="#" className="hover:text-indigo-600 transition-colors">Billing</a>
          </div>
        </div>
      </footer>
    </div>
  );
};

export default App;
